CREATE PROCEDURE [dbo].[ESP_SAMPLE_LSJ_UPDATE]
(
	@COM_CODE      		VARCHAR(6)
,	@CODE				VARCHAR(5)
,	@CODE_NAME			VARCHAR(20)		= NULL
,	@BUSINESS_NO		VARCHAR(30)		= NULL
,	@QTY				NUMERIC(28,10)	= NULL
,	@USE_YN				CHAR(1)			= NULL
,	@MODIFY_ID			VARCHAR(30)
)
AS

SET NOCOUNT ON  
SET TRANSACTION ISOLATION LEVEL READ UNCOMMITTED 

UPDATE SAMPLE_LSJ 
SET 
	COM_CODE=ISNULL(@COM_CODE,COM_CODE),
	CODE_NAME = ISNULL(@CODE_NAME, CODE_NAME),
	BUSINESS_NO = ISNULL(@BUSINESS_NO,BUSINESS_NO),
	QTY = ISNULL(@QTY, QTY),
	USE_YN = ISNULL(@USE_YN,USE_YN),
	WRITE_ID = WRITE_ID,
	WRITE_DT = WRITE_DT,
	MODIFY_ID = ISNULL(@MODIFY_ID,MODIFY_ID),
	MODIFY_DT = GETDATE()
WHERE CODE = @CODE

SET NOCOUNT OFF
